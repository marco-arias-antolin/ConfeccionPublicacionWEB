    function Subir() {
      var formulario = "";
      var usuario = document.getElementById('usuario').value;
      var nombre = document.getElementById('nombre').value;
      var apellido = document.getElementById('apellido').value;
      var email = document.getElementById('email').value;
      var clave1 = document.getElementById('clave1').value;
      var clave2 = document.getElementById('clave2').value;
      var hobbies = [];
      var hobby = document.querySelectorAll('input[name="hobby"]:checked');
      for (let node of hobby.values()) {
        hobbies.push(node.value);
      }
      hobbies = hobbies.join(', ');
      var pais = document.getElementById('pais').value;
      var sexo = document.querySelector('input[name="sexo"]:checked');
      sexo = sexo ? sexo.value : 'n';
      if (usuario == "" || nombre == "" || apellido == "" || pais == "") {
        alert("Los campos con asterisco son obligatorios!!!");
        return false;
      }
      if (hobbies.length == 0) {
        alert("Debes seleccionar al menos un hobby!!");
        return false;
      }
      if ((clave1 != clave2) || (clave1 == "")) {
        alert("Las claves deben ser iguales y no pueden ser en blanco!!!");
        return false;
      }
      if (email == "") email = "No informado";
      if (sexo == "" || sexo == "n") sexo = "No informado";
      formulario = `Usuario: ${usuario}
            Nombre: ${nombre}
            Apellido: ${apellido}
            E-mail: ${email}
            Clave: ${clave1}
            Hobbies: ${hobbies}
            Pais: ${pais}
            Sexo: ${sexo}`;
      formulario = formulario.replace(/\t/g, '');
      alert(formulario);
      return true;
    }
  </script>

El script contiene variables, estructuras de control y funciones para realizar operaciones de manipulación, entrada y salida de datos en el formulario. A continuación se describen los diferentes aspectos del código:

Variables:

  • formulario: Almacena el contenido del formulario que se mostrará en el mensaje de alerta.
  • usuario, nombre, apellido, email, clave1, clave2: Variables que almacenan los valores ingresados en los campos de texto del formulario.
  • hobbies: Un array que almacena los valores de los hobbies seleccionados en el formulario.
  • b: Variable no definida en el código original.
  • a: Variable utilizada en el bucle for para recorrer los elementos del formulario.

Estructuras de programación:

  • Función Subir(): Es la función principal del script y se ejecuta cuando se hace clic en el botón "Crear usuario". Realiza las siguientes acciones:
  • Obtiene los valores ingresados en los campos del formulario utilizando los métodos getElementById y querySelectorAll.
  • Realiza diversas comprobaciones para validar los datos ingresados en el formulario, como verificar que los campos obligatorios no estén vacíos, que se haya seleccionado al menos un hobby y que las claves coincidan.
  • Construye la cadena formulario con los valores ingresados en el formulario.
  • Muestra el contenido del formulario en un mensaje de alerta.
  • Devuelve true si todas las validaciones son exitosas y el formulario se puede enviar.

Instrucciones de manipulación, entrada y salida de datos:

  • document.getElementById('id'): Obtiene el elemento del DOM con el ID especificado.
  • document.querySelectorAll('selector'): Obtiene una lista de elementos del DOM que coinciden con el selector especificado.
  • document.form1.hobby[a].checked: Verifica si un checkbox de hobby está seleccionado.
  • element.value: Obtiene el valor de un elemento del formulario.
  • element.values(): Devuelve un objeto iterable de los valores de los elementos seleccionados.
  • element.join(', '): Combina los elementos de un array en una cadena separada por comas y espacios.
  • element.replace(/\t/g, ''): Elimina las tabulaciones de la cadena formulario.
